class CoffeeMachine {
	void MakeCoffee() {
		setBoiler(true);
		setWarmer(true);
		setValve(true);
		setLight(true);
		while (getPlate() != 3){
			//how to say keep making coffee?
		}
		setBoiler(false);
		setWarmer(false);
		setValve(false);
		setLight(false);
	}

	void setBoiler(bool turnOn) {
		if (turnOn) {
			myBoiler.turnOn();
		}
		else 
			myBoiler.turnOff();
	}
}

class Boiler {

	}

		myIsTurnedOn = false;
	}
