class CoffeeMachine {
	void MakeCoffee() {
		setBoiler(true);
		setWarmer(true);
		setValve(true);
	}

	void setBoiler(bool turnOn) {
		if (turnOn) {
			myBoiler.turnOn();
		}
		else 
			myBoiler.turnOff();
	}
}

class Boiler {
	void turnOn(){
		myIsTurnedOn = true;
	}
	void turnOff(){
		myIsTurnedOn = false;
	}
}
